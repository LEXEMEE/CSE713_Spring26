import os

import pandas as pd

import seaborn as sns
from PIL import Image
import imagehash
from sklearn.model_selection import train_test_split

from sklearn.metrics import classification_report, confusion_matrix
import tensorflow as tf

# =========================================================
# 5. SET DATASET ROOT
# =========================================================
DATA_ROOT = path

# =========================================================
# 6. DEFINE VALID IMAGE FORMATS AND LABEL ALIASES
# =========================================================
IMAGE_EXTS = {".jpg", ".jpeg", ".png", ".bmp", ".tif", ".tiff"}
LABEL_ALIASES = {
    "glioma": ["glioma"],
    "meningioma": ["meningioma"],
    "pituitary": ["pituitary"],
    "no_tumor": ["no_tumor", "notumor", "no tumor", "no-tumor", "normal", "healthy"]
}
# =========================================================
# 7. FUNCTION TO INFER LABEL FROM FILE PATH
# =========================================================
def infer_label_from_path(path):
    path_lower = path.lower().replace("\\", "/")
    for label, aliases in LABEL_ALIASES.items():
        for alias in aliases:
            if alias in path_lower:
                return label
    return None
# =========================================================
# 8. SCAN DATASET AND BUILD DATAFRAME
# =========================================================
rows = []
for root, dirs, files in os.walk(DATA_ROOT):
    for fname in files:
        ext = os.path.splitext(fname)[1].lower()
        if ext in IMAGE_EXTS:
            fpath = os.path.join(root, fname)
            label = infer_label_from_path(fpath)
            if label is not None:
                rows.append({
                    "filepath": fpath,
                    "label": label
                })

def compute_phash(path):

plt.figure(figsize=(8, 5))
sns.countplot(data=df_dedup, x="label", order=df_dedup["label"].value_counts().index)

train_df, temp_df = train_test_split(

)
val_df, test_df = train_test_split(
    temp_df,
    test_size=0.50,
    random_state=42,
    stratify=temp_df["label"]
)
print("Train size:", len(train_df))

print("Test size:", len(test_df))
# =========================================================
# 15. VISUALIZE SPLIT DISTRIBUTION

# =========================================================
# 16. CLAHE PREPROCESSING
# =========================================================
def clahe_rgb(img_rgb):
    lab = cv2.cvtColor(img_rgb, cv2.COLOR_RGB2LAB)
    l, a, b = cv2.split(lab)
    clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8, 8))
    l2 = clahe.apply(l)
    lab2 = cv2.merge((l2, a, b))
    return cv2.cvtColor(lab2, cv2.COLOR_LAB2RGB)
# =========================================================
# 17. COMPARE ORIGINAL AND CLAHE IMAGE
# =========================================================
example_row = train_df.sample(1, random_state=1).iloc[0]

# =========================================================
# 18. DEFINE CLASS MAPPINGS
# =========================================================
CLASS_NAMES = ["glioma", "meningioma", "pituitary", "no_tumor"]
class_to_idx = {name: i for i, name in enumerate(CLASS_NAMES)}
idx_to_class = {i: name for name, i in class_to_idx.items()}
print(class_to_idx)
# =========================================================
# 19. SET PIPELINE PARAMETERS
# =========================================================
IMG_SIZE = 224
BATCH_SIZE = 32
AUTOTUNE = tf.data.AUTOTUNE
# =========================================================
# 20. LOAD AND PREPROCESS FUNCTION
# =========================================================
def load_and_preprocess(path, label):
    path = path.numpy().decode("utf-8")
    label = label.numpy().decode("utf-8")
    img = cv2.imread(path)
    img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
    img = clahe_rgb(img)
    img = cv2.resize(img, (IMG_SIZE, IMG_SIZE))
    img = img.astype("float32")
    img = preprocess_input(img)
    label_idx = np.int32(class_to_idx[label])
    return img, label_idx
# =========================================================
# 21. TENSORFLOW WRAPPER
# =========================================================
def tf_wrapper(path, label):
    img, label_idx = tf.py_function(
        func=load_and_preprocess,
        inp=[path, label],
        Tout=[tf.float32, tf.int32]
    )
    img.set_shape((IMG_SIZE, IMG_SIZE, 3))
    label_idx.set_shape(())
    return img, label_idx
# =========================================================
# 22. CREATE TENSORFLOW DATASETS
# =========================================================
def make_dataset(df_part, training=False):
    ds = tf.data.Dataset.from_tensor_slices((
        df_part["filepath"].values,
        df_part["label"].values
    ))
    ds = ds.map(tf_wrapper, num_parallel_calls=AUTOTUNE)
    if training:
        ds = ds.shuffle(1024, reshuffle_each_iteration=True)
    ds = ds.batch(BATCH_SIZE).prefetch(AUTOTUNE)
    return ds
train_ds = make_dataset(train_df, training=True)
val_ds = make_dataset(val_df, training=False)
test_ds = make_dataset(test_df, training=False)
# =========================================================
# 23. SHOW PROCESSED IMAGES
# =========================================================
images, labels = next(iter(train_ds))
plt.figure(figsize=(14, 10))
for i in range(min(12, len(images))):
    img_to_show = images[i].numpy().copy()
    img_to_show = img_to_show - img_to_show.min()
    img_to_show = img_to_show / (img_to_show.max() + 1e-8)
    plt.subplot(3, 4, i + 1)
    plt.imshow(img_to_show)
    plt.title(idx_to_class[int(labels[i].numpy())])
    plt.axis("off")
plt.tight_layout()
plt.show()
# =========================================================
# 24. COMPUTE CLASS WEIGHTS
# =========================================================
y_train = train_df["label"].map(class_to_idx).values
class_weights_values = compute_class_weight(
    class_weight="balanced",
    classes=np.array(list(range(len(CLASS_NAMES)))),
    y=y_train
)
class_weights = {i: w for i, w in enumerate(class_weights_values)}
print(class_weights)
# =========================================================
# 25. DATA AUGMENTATION
# =========================================================
data_augmentation = keras.Sequential([
    layers.RandomFlip("horizontal"),
    layers.RandomRotation(0.03),
    layers.RandomZoom(0.08),
    layers.RandomContrast(0.08),
    layers.GaussianNoise(0.02),
], name="augmentation")
# =========================================================
# 26. BUILD IMPROVED EFFICIENTNET MODEL
# =========================================================
base_model = keras.applications.EfficientNetB0(
    include_top=False,
    weights="imagenet",
    input_shape=(IMG_SIZE, IMG_SIZE, 3)
)
base_model.trainable = False
inputs = keras.Input(shape=(IMG_SIZE, IMG_SIZE, 3))
x = data_augmentation(inputs)
x = base_model(x, training=False)
x = layers.GlobalAveragePooling2D(name="global_average_pooling2d")(x)
x = layers.BatchNormalization(name="batch_normalization")(x)
x = layers.Dense(256, activation="relu", name="dense")(x)
x = layers.Dropout(0.4, name="dropout")(x)
outputs = layers.Dense(len(CLASS_NAMES), activation="softmax", name="dense_1")(x)
model = keras.Model(inputs, outputs)
# =========================================================
# 27. COMPILE MODEL
# =========================================================
model.compile(
    optimizer=keras.optimizers.Adam(learning_rate=3e-4, clipnorm=1.0),
    loss=tf.keras.losses.SparseCategoricalCrossentropy(),
    metrics=["accuracy"]
)
model.summary()
# =========================================================
# 28. CALLBACKS
# =========================================================
callbacks = [
    keras.callbacks.EarlyStopping(
        monitor="val_loss",
        patience=5,
        restore_best_weights=True
    ),
    keras.callbacks.ReduceLROnPlateau(
        monitor="val_loss",
        factor=0.5,
        patience=2,
        min_lr=1e-7,
        verbose=1
    ),
    keras.callbacks.ModelCheckpoint(
        "best_brain_tumor_model.keras",
        monitor="val_loss",
        save_best_only=True,
        verbose=1
    )
]
# =========================================================
# 29. TRAIN STAGE 1
# =========================================================
history1 = model.fit(
    train_ds,
    validation_data=val_ds,
    epochs=12,
    class_weight=class_weights,
    callbacks=callbacks
)
# =========================================================
# 30. FINE-TUNE STAGE 2
# =========================================================
base_model.trainable = True
for layer in base_model.layers[:-20]:
    layer.trainable = False
model.compile(
    optimizer=keras.optimizers.Adam(learning_rate=1e-6, clipnorm=1.0),
    loss=tf.keras.losses.SparseCategoricalCrossentropy(),
    metrics=["accuracy"]
)
history2 = model.fit(
    train_ds,
    validation_data=val_ds,
    epochs=10,
    class_weight=class_weights,
    callbacks=callbacks
)
# =========================================================
# 31. PLOT TRAINING HISTORY
# =========================================================
acc = history1.history["accuracy"] + history2.history["accuracy"]
val_acc = history1.history["val_accuracy"] + history2.history["val_accuracy"]
loss = history1.history["loss"] + history2.history["loss"]
val_loss = history1.history["val_loss"] + history2.history["val_loss"]
epochs_range = range(1, len(acc) + 1)
plt.figure(figsize=(14, 5))
plt.subplot(1, 2, 1)
plt.plot(epochs_range, acc, label="Train Accuracy")
plt.plot(epochs_range, val_acc, label="Validation Accuracy")
plt.title("Training vs Validation Accuracy")
plt.xlabel("Epoch")
plt.ylabel("Accuracy")
plt.legend()
plt.subplot(1, 2, 2)
plt.plot(epochs_range, loss, label="Train Loss")
plt.plot(epochs_range, val_loss, label="Validation Loss")
plt.title("Training vs Validation Loss")
plt.xlabel("Epoch")
plt.ylabel("Loss")
plt.legend()
plt.tight_layout()
plt.savefig("training_curves.png", dpi=300, bbox_inches="tight")

# =========================================================
# 32. LOAD BEST MODEL AND EVALUATE
# =========================================================
model = keras.models.load_model("best_brain_tumor_model.keras")
test_loss, test_acc = model.evaluate(test_ds)
print("Test Loss:", test_loss)
print("Test Accuracy:", test_acc)
# =========================================================
# 33. PREDICT ON TEST SET
# =========================================================
y_true = []
y_pred = []
y_prob = []
for batch_images, batch_labels in test_ds:
    preds = model.predict(batch_images, verbose=0)
    pred_classes = np.argmax(preds, axis=1)
    y_true.extend(batch_labels.numpy())
    y_pred.extend(pred_classes)
    y_prob.extend(np.max(preds, axis=1))
y_true = np.array(y_true)
y_pred = np.array(y_pred)
y_prob = np.array(y_prob)

# =========================================================
# 38. GRAD-CAM FUNCTION
# =========================================================
def make_gradcam_heatmap(img_array, pred_index=None):
    with tf.GradientTape() as tape:
        conv_outputs = last_conv_layer_model(img_array, training=False)
        tape.watch(conv_outputs)
        preds = classifier_model(conv_outputs, training=False)
        if pred_index is None:
            pred_index = tf.argmax(preds[0])
        class_channel = preds[:, pred_index]
    grads = tape.gradient(class_channel, conv_outputs)
    pooled_grads = tf.reduce_mean(grads, axis=(0, 1, 2))
    conv_outputs = conv_outputs[0]
    heatmap = tf.reduce_sum(conv_outputs * pooled_grads, axis=-1)
    heatmap = tf.maximum(heatmap, 0)
    heatmap /= (tf.reduce_max(heatmap) + 1e-8)
    return heatmap.numpy()
